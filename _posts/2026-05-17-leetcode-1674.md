The problem requires to return the minimum number of replacement as an `int` value to make the array `nums` complementary.

> DISCLAIMER: Below solution spoiler is available.
{: .prompt-warning }

## Idea 1
Thinking the greedy algorithm, the most frequent sum value of `nums[i] + nums[n-1-i]` would lead to the minimum replacements.

The idea had failed to handle all edge cases well, so it was discarded later.

### Trial 1
> DISCLAIMER: The trial fail some test cases.
{: .prompt-danger }

The Code 1-1 and Code 1-2 fail some cases where the most frequent sums do not make the minimum number of replacements. While I tried to handle some cases where all of the most frequent sums make more replacements than the sum value that does not derive from any pairs from `sums`. However it fails some other many edge cases while making the code more complex.

```cpp
class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        const int n = nums.size();
        vector<int> vec_sum(limit * 2 + 1, 0);

        // Find the sum with the most counts
        for(int i=0; i<n/2; ++i){
            ++vec_sum[nums[i] + nums[n-1-i]];
        }

        stack<int> max_nums;
        int max_count = 0;

        for(int i=0; i<vec_sum.size(); ++i){
            int s = vec_sum[i];
            if(s > max_count){
                max_nums = stack<int>();
                max_nums.push(i);
                max_count = s;
            }
            else if(s == max_count){
                max_nums.push(i);
            }
        }


        int answer = numeric_limits<int>::max();
        // Compare the pairs to count the replacements
        while(!max_nums.empty()){
            int k = max_nums.top();
            max_nums.pop();
            int rep_count = 0;

            for(int i=0; i<n/2; ++i){
                int s = nums[i] + nums[n-1-i];
                
                if(k != s){

                    int flag = 0;

                    int a = k - nums[i];
                    int b = k - nums[n-1-i];

                    if(a >= 1 && a <= limit){
                        flag |= 1;
                    }
                    if(b >= 1 && b <= limit){
                        flag |= 1;
                    }

                    rep_count += (2 - flag);
                }
            }

            if(answer > rep_count)
                answer = rep_count;
        }
        return answer;
    }
};
```
