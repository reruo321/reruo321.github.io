## Idea
Since all robots have the same speed and unique initial position, any pairs of robots would never meet unless they are moving to each other. So the exact positions of robots do not matter, but we just need the relationship between two nearby robots.

## Algorithm
I sorted robots used `stack`.

```cpp
struct Robot {
    int no;
    int pos;
    int hp;
    char dir;

    Robot(int no, int pos, int hp, char dir): no(no), pos(pos), hp(hp), dir(dir) {}
    Robot(): no(0), pos(0), hp(0), dir('L') {}
};

bool pos_comp(const Robot &a, const Robot &b) {
    return a.pos < b.pos;
}

bool no_comp(const Robot &a, const Robot &b){
    return a.no < b.no;
}

class Solution {
public:
    vector<int> survivedRobotsHealths(vector<int>& positions, vector<int>& healths, string directions) {
        vector<Robot> r_vec;
        const int n = positions.size();

        r_vec.assign(n, Robot());

        for(int i=0; i<n; ++i){
            r_vec[i] = Robot(i, positions[i], healths[i], directions[i]);
        }
        sort(r_vec.begin(), r_vec.end(), pos_comp);

        stack<Robot> r_stack;

        for(int i=0; i<n; ++i){
            if(r_stack.empty()){
                r_stack.push(r_vec[i]);
            }
            else{
                Robot right = r_vec[i];
                bool ispush = false;

                while(1){
                    if(!r_stack.empty()) {
                        // Collide
                        Robot left = r_stack.top();

                        if(left.dir == 'R' && right.dir == 'L'){
                            r_stack.pop();
                            // Left
                            if(left.hp > right.hp){
                                r_stack.push(Robot(left.no, left.pos, left.hp - 1, left.dir));
                                break;
                            }

                            // Right
                            else if(left.hp < right.hp){
                                --right.hp;
                                continue;
                            }

                            // None
                            else {
                                break;
                            }
                        }
                    }
                    r_stack.push(right);
                    break;
                    // end of while
                }
            }
        }
        vector<Robot> final_robot_vec;

        while(!r_stack.empty()){
            Robot t = r_stack.top();
            final_robot_vec.push_back(t);
            r_stack.pop();
        }
        sort(final_robot_vec.begin(), final_robot_vec.end(), no_comp);
        vector<int> answer_vec;

        for(auto &r: final_robot_vec){
            answer_vec.push_back(r.hp);
        }
        return answer_vec;
    }
};
```

```cpp
        vector<int> hp_vec(n, -1);
        vector<int> idx_vst;
        idx_vst.reserve(n);
        int cur = 0;

        for(int i: indices){
            if(directions[i] == 'R'){
                idx_vst.push_back(i);
            }
            else{
                while(!idx_vst.empty() && directions[idx_vst.back()] == 'R' && healths[i] > 0){
                    if(healths[idx_vst.back()] < healths[i]){
                        --healths[i];
                        idx_vst.pop_back();
                    }
                    else if(healths[idx_vst.back()] > healths[i]){
                        --healths[idx_vst.back()];
                        healths[i] = 0;
                    }
                    else{
                        idx_vst.pop_back();
                        healths[i] = 0;
                    }
                }
                if(healths[i] > 0){
                    idx_vst.push_back(i);
                }
            }
        }

        int k = idx_vst.size();
        vector<int> answer;
        answer.reserve(k);

        for(int i: idx_vst){
            hp_vec[i] = healths[i];
        }

        for(int h: hp_vec){
            if(h > 0){
                answer.push_back(h);
            }
            if(answer.size() == k)
                break;
        }

        return answer;
    }
};
```
